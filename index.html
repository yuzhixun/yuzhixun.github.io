升序排列和降序排列

using namespace std;

int main(){
    int f[10] = {0,9,2,8,3,4,8,-9,-2,1};
    sort(f,f+10);//升序排列
    for(int i = 0;i<10;i++){
        cout << f[i] << ' ';
    } 
    return 0;
}

#include <bits/stdc++.h>

using namespace std;

int main(){
    int f[10] = {0,9,2,8,3,4,8,-9,-2,1};
    sort(f,f+10,greater<int>()); //降序排列

        cout << f[i] << ' ';
    } 
    return 0;
}

三个参数，元素类型，储存元素的容器，以及排列的方式

#include <bits/stdc++.h>
using namespace std;

int main()
{
  ios::sync_with_stdio(0),cin.tie(0),cout.tie(0);
  priority_queue<int,vector<int>,greater<int>>pq;
  string p;
  int n;int x;
  cin>>n;
  while(n--){
    cin>>p;
    if(p=="push"){
      cin>>x;
      pq.push(x);
    }else if(p=="remove"){
      if(!pq.empty()){
        pq.pop();
      }else{
        cout<<"empty"<<endl;
      }

    }else if(p=="min"){
      if(!pq.empty()){
        cout<<pq.top()<<endl;
      }else{
        cout<<"empty"<<endl;
      }

    }else if(p=="print"){
      int k;
      cin>>k;
      while(k--){
        cout<<pq.top()<<" ";
        pq.pop();
      }
      cout<<endl;
    }
  }
 
  return 0;
}

查找字符在字符串中出现的位置

#include <iostream>
#include <string>

using namespace std;

int main() {
    string str = "Hello, World!";
    char ch = 'o';
    // 查找字符 'o' 在字符串中的位置
    size_t pos = str.find(ch);

    if (pos != string::npos) {
        cout << "字符 '" << ch << "' 在字符串中的位置为：" << pos << endl;
    } else {
        cout << "字符 '" << ch << "' 未找到！" << endl;
    }
    
    return 0;
}
