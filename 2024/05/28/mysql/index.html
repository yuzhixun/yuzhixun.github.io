一些重要的代码

show databases;

use --数据库

show tables; --显示你库中所有的表
describe --显示表中的内容

create database --库名 --创建数据库
exit --推出连接

数据库的数据类型

数据库的字段属性

CREATE TABLE IF NOT EXISTS `student` (
    id INT(4) NOT NULL AUTO_INCREMENT COMMENT '学号',
    name VARCHAR(20) NOT NULL DEFAULT '匿名' COMMENT '姓名',
    pwd VARCHAR(20) NOT NULL DEFAULT '123456' COMMENT '密码',
    sex VARCHAR(2) NOT NULL DEFAULT '女' COMMENT '性别',
    address VARCHAR(100) DEFAULT NULL COMMENT '家庭住址',
    email VARCHAR(50) DEFAULT NULL COMMENT '邮箱',
    PRIMARY KEY (id)
);

dql

select studentno,studentname from student;//查询某张表里面的某类字段的值
select studentno as 学号,studentname as 姓名 from student;//查询某张表里面的某类字段的值，并给这个字段设置名字
select concat("姓名：",studentname) as 新姓名 from student;//concat（）函数，将两个字符串拼接起来

条件语句

select studentno,subjectno from result where studentresult between 80 and 90;//给查找增加条件 
select studentno,studentresult from result where studentno != 1000;
select studentno,studentresult from result where not studentno = 1000;

模糊查询

 select studentno,studentname from result where studentname like '刘%';//%后可以有很多字符，查询信刘的人
 select studentno,studentname from result where studentname like '刘_';//_接一个字符
  select studentno,studentname from result where studentname like '刘__';//接两个字符
 select studentno,studentname from result where studentname like '%二%';//中间有二字的
